import java.util.*;
public class  Main{
    public static void main(String [] args){
       Scanner sc=new Scanner(System.in) ;
        String s= sc.nextLine();
        s=s.toLowerCase();
ArrayList<Character> li= new ArrayList<>();
for (int i=0;i<s.length();i++)
     {
    char ch=s.charAt(i);
    if(!li.contains(ch))
    {
        li.add(ch);
    }

        }
        System.out.println(li.toString());

    }
}
